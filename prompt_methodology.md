```kotlin
@Service
class OrderService {
    fun processOrders() {
        val orders = orderRepository.findAll()
        for (order in orders) {

                if (user.active) {

            }
        }
    }
}
```

```kotlin
@Service
class OrderService(
    private val promoCodeRepository: PromoCodeRepository,
    private val discountCalculator: DiscountCalculator
) {
    fun applyPromoCode(orderId: Long, code: String): BigDecimal {
        val promo = promoCodeRepository.findByCode(code)
            ?: throw PromoCodeNotFoundException(code)
        if (promo.expiresAt.isBefore(LocalDateTime.now())) {
            throw PromoCodeExpiredException(code)
        }
        return discountCalculator.calculate(orderId, promo.discountPercent)
    }
}
```
